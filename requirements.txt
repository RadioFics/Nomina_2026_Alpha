# ============================================================================
#  INSTALACION EN AZURE APP SERVICE WINDOWS (Python 3.6.8 via "py -3"):
#
#  PASO 1 — Crear carpeta de destino y fijar PYTHONPATH en App Settings:
#    En Azure Portal → Configuration → Application settings:
#      PYTHONPATH = D:\home\site\pythonpkgs
#      PYTHON_PATH = C:\Windows\py.exe
#      PYTHON_ARGS = -3
#
#  PASO 2 — Instalar en Kudu CMD (https://<app>.scm.azurewebsites.net/DebugConsole):
#    mkdir D:\home\site\pythonpkgs
#    py -3 -m pip install --target D:\home\site\pythonpkgs "pypdf==4.3.1"
#    py -3 -m pip install --target D:\home\site\pythonpkgs "Pillow>=8.0.0,<9.0.0"
#    py -3 -m pip install --target D:\home\site\pythonpkgs "pytesseract==0.3.7"
#    py -3 -m pip install --target D:\home\site\pythonpkgs --no-deps "pdfplumber==0.5.28"
#    py -3 -m pip install --target D:\home\site\pythonpkgs --no-deps "pdfminer.six==20200517"
#    py -3 -m pip install --target D:\home\site\pythonpkgs "chardet==5.0.0" "sortedcontainers==2.4.0"
#    py -3 -m pip install --target D:\home\site\pythonpkgs "Wand==0.6.13"
#
#  PASO 3 — Verificar en Kudu:
#    py -3 -c "import sys; sys.path.insert(0,'D:\\home\\site\\pythonpkgs'); import pdfplumber; print('OK')"
#
#  NOTA sobre reportlab: solo disponible como fuente para Python 3.6 (requiere
#  compilador MSVC que no existe en Azure App Service). Se instala localmente
#  para desarrollo; en producción se omite o se usa un servidor con compilador.
# ============================================================================
